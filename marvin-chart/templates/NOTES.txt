✅ Post-deployment verification checklist:
   1. Check pod status:
      oc get pods -l app.kubernetes.io/name={{ include "marvin.name" . }}

   2. Verify health endpoint:
{{- if .Values.route.enabled }}
      curl -v https://$ROUTE/healthz
{{- else }}
      oc port-forward svc/{{ include "marvin.fullname" . }} 8080:{{ .Values.service.port }}
      curl -v http://localhost:8080/healthz
{{- end }}

   3. Check scheduler state file:
      oc exec $POD -- cat {{ .Values.config.dataDir }}/scheduler_state.json

   4. Verify no tokens in logs:
      oc logs $POD | grep -i "token.*:" | head -20
      (Should NOT show plaintext JWTs)

   5. Check migration job logs:
      MIGRATION_JOB=$(oc get jobs -l job-type=migration --sort-by=.metadata.creationTimestamp -o jsonpath='{.items[-1].metadata.name}')
      oc logs job/$MIGRATION_JOB
